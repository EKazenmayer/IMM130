        public class JavaCrashCourse {
            public static void main(String[] args) {
                Barista barista = new Barista();
                barista.setName("yourName");
                barista.setAge(yourAge);
                barista.printWelcomeMessage();
                
                boolean drinksCoffee = yourAnswer;
                int packetsOfSugar = yourNum;
                makeDrink(drinksCoffee, packetsOfSugar);
            }
            
            public static void makeDrink(boolean isCoffee, int numPackets) {
                String typeOfDrink;
                if (isCoffee = true) {
                    typeOfDrink = "coffee";
                }
                else {
                    typeOfDrink = "tea";
                }
                System.out.print("Barista Ada made you a special drink! ");
                System.out.println("She made you a " + typeOfDrink + " with " + numPackets + " of sugar.");
                
                if (numPackets == 0) {
                    System.out.println("It's really bitter!");
                }
                else if (numPackets == 1 || numPackets == 2) {
                    System.out.println("Just right!");
                }
                else if (numPackets >= 3) {
                    System.out.println("How sweet!");
                }
            }
        }
