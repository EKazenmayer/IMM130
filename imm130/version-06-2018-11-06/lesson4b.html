How It Works

int amountMoreCups = yourInput;
int numberOfCustomers = customers.size();
int numberOfCups = 2;
for(int customer = 1; customer < numberOfCustomers + 1; customer = customer + 1) {
    System.out.println("Customer #" + customer + " ordered " + numberOfCups + " cups of coffee.");
    numberOfCups = numberOfCups + amountMoreCups;
}
