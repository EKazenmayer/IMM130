How It Works

        public class JavaCrashCourse {
            public static void main(String[] args) {
                Barista barista = new Barista();
                barista.setName("yourName");
                barista.setAge(yourAge);
                barista.printWelcomeMessage();
            }
        }
        
        public class Barista {
            String name;
            int age;
            
            public Barista() {
                name = "";
                age = 0;
            }
            
            public void setName(String yourName) {
                name = yourName;
            }
            
            public void setAge(int yourAge) {
                age = yourAge;
            }
            
            public void printWelcomeMessage() {
                System.out.println("Welcome to the Cafe, " + name + "!");
                System.out.print("You're " + age + "? Cool beans! ");
                System.out.println("In fact, if today was your birthday, we would give you "
                + age + " coffee beans - one for each year!");
            }
        }
